\section{Overlay Commands}

% Demonstrating different overlay commands
\begin{frame}{Overlay Commands Comparison}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Using \textbackslash onslide:}
            \begin{itemize}
                \item<1-> Always reserves space
                \item<2-> Content invisible but present
                \item<3-> Layout stays consistent
            \end{itemize}
        \end{column}

        \begin{column}{0.48\textwidth}
            \textbf{Using \textbackslash only:}
            \begin{itemize}
                \item<1-> No space when hidden
                \item<2-> Layout can shift
                \item<3-> Useful for replacements
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

% Complex overlay example
\begin{frame}{Complex Overlay Specifications}
    \begin{block}{Block 1}
        \onslide<1->{This block is always visible.}
    \end{block}

    \begin{alertblock}{Alert Block}<2->
        This appears on slide 2 onwards.
    \end{alertblock}

    \begin{exampleblock}{Example Block}<3>
        This appears only on slide 3 (not on 4+).
    \end{exampleblock}

    \onslide<4->{
        \begin{block}{Final Block}
            This appears on slide 4 onwards.
        \end{block}
    }
\end{frame}

\section{Best Practices}

% Good overlay usage
\begin{frame}{Best Practices for Overlays}
    \textbf{Do:}
    \begin{itemize}
        \item<1-> Use overlays to guide audience focus
        \item<2-> Keep overlay count ≤5 per frame
        \item<3-> Use consistent reveal patterns
        \item<4-> Test visibility at target screen size
    \end{itemize}

    \medskip

    \onslide<5->{
        \textbf{Don't:}
        \begin{itemize}
            \item Overuse overlays (cognitive overload)
            \item Create layout shifts with \textbackslash only
            \item Forget about handout mode
        \end{itemize}
    }
\end{frame}

% Accessibility considerations
\begin{frame}{Accessibility Considerations}
    \begin{block}{WCAG 2.1 Contrast Requirements}
        \begin{itemize}
            \item<1-> Normal text: ≥4.5:1 contrast ratio (AA)
            \item<2-> Large text (≥18pt): ≥3:1 contrast ratio (AA)
            \item<3-> Test with online contrast checkers
        \end{itemize}
    \end{block}

    \onslide<4->{
        \begin{alertblock}{Tip}
            Avoid relying solely on color to convey information.
            Use shapes, patterns, or labels in addition to color.
        \end{alertblock}
    }
\end{frame}
